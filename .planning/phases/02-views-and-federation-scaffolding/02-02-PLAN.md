    **1. Create `packages/ui/src/components/TerminalPanel.vue`:**

    Uses `<script setup lang="ts">`. Receives all data via props (NO `@/` imports -- packages/ui independence rule).

    **Props interface:**
    ```typescript
    interface TerminalEntry {
      input: string
      output: string
    }

    interface CommandDef {
      description: string
      output: string
    }

    const { defaultOutput = [], commands = {} } = defineProps<{
      defaultOutput?: TerminalEntry[]
      commands?: Record<string, CommandDef>
    }>()
    ```

    **Reactive state:**
    ```typescript
    import { ref, nextTick, onMounted } from 'vue'

    const history = ref<TerminalEntry[]>([...defaultOutput])
    const currentInput = ref('')
    const historyEnd = ref<HTMLElement | null>(null)
    ```

    **Command execution function:**
    ```typescript
    function executeCommand(input: string): string {
      const [cmd, ...args] = input.trim().split(' ')
      if (cmd === 'ls') {
        return Object.entries(commands)
          .map(([name, def]) => `${name} -- ${def.description}`)
          .join('\n')
      }
      if (cmd === 'help' && args[0]) {
        const target = commands[args[0]]
        return target ? target.description : `Unknown command: ${args[0]}`
      }
      const entry = commands[cmd]
      if (entry) return entry.output
      return `command not found: ${cmd}. Type 'ls' to see available commands.`
    }
    ```

    **Submit handler:**
    ```typescript
    async function handleSubmit() {
      const input = currentInput.value.trim()
      if (!input) return
      const output = executeCommand(input)
      history.value.push({ input, output })
      currentInput.value = ''
      await nextTick()
      historyEnd.value?.scrollIntoView({ behavior: 'smooth' })
    }
    ```

    **onMounted:** Scroll to bottom of history on mount.

    **Template structure:**
    ```
    <div class="bg-surface-raised rounded-lg border border-border p-8"
         style="box-shadow: 0 0 15px rgba(114, 241, 184, 0.15)">
      <!-- History area: overflow-y-auto, max-h-[400px] -->
      <div class="max-h-[400px] overflow-y-auto font-mono text-sm">
        <div v-for="(entry, i) in history" :key="i" class="mb-3">
          <!-- Input line: $ prompt in bold accent pink, command text in text color -->
          <div class="flex gap-2">
            <span class="font-bold text-accent">$</span>
            <span class="text-text">{{ entry.input }}</span>
          </div>
          <!-- Output: whitespace-pre-wrap, text-accent-cyan for normal output,
               text-destructive for "command not found" errors -->
          <div class="ml-5 mt-1 whitespace-pre-wrap"
               :class="entry.output.startsWith('command not found') ? 'text-destructive' : 'text-accent-cyan'">
            {{ entry.output }}
          </div>
        </div>
        <div ref="historyEnd" />
      </div>
      <!-- Input area: form with text input -->
      <form @submit.prevent="handleSubmit" class="mt-4 flex items-center gap-2 font-mono text-sm">
        <span class="font-bold text-accent">$</span>
        <input
          v-model="currentInput"
          type="text"
          class="flex-1 bg-transparent text-text outline-none placeholder:text-text-muted"
          placeholder="Type a command..."
          autofocus
        />
        <!-- Blinking cursor indicator -->
        <span class="inline-block h-4 w-2 bg-accent-yellow animate-[blink_1s_step-end_infinite]" />
      </form>
    </div>
    ```

    The terminal panel border glow uses inline `box-shadow` style per UI-SPEC: `0 0 15px rgba(114, 241, 184, 0.15)` (subtle cyan glow).
