**useTheme.test.ts:**
Because useTheme calls `useThemeStore()` internally, it MUST be invoked inside a Vue component that has Pinia available. Use the wrapper component pattern from the RESEARCH.md:

```typescript
import { describe, it, expect } from 'vitest'
import { mount } from '@vue/test-utils'
import { createTestingPinia } from '@pinia/testing'
import { useTheme } from '@/composables/useTheme'

describe('useTheme', () => {
  it('applies CSS variables to document.documentElement on init', () => {
    const TestComponent = {
      setup() {
        useTheme()
        return {}
      },
      template: '<div />',
    }

    mount(TestComponent, {
      global: { plugins: [createTestingPinia()] },
    })

    const root = document.documentElement.style
    expect(root.getPropertyValue('--color-surface')).toBeTruthy()
  })
})
```

Test all 16 CSS variables are set. The default theme is synthwave-84, so verify:

- `--color-surface` = '#262335'
- `--color-accent` = '#ff7edb'
- (and all other 14 variables have truthy values)

Also test that the return value is the theme store (has `setTheme` function).
