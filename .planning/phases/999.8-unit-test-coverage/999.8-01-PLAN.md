1. Install root-level dev dependencies:

2. Create `vitest.config.ts` at project root with content:

   ```typescript

     test: {
       projects: ['apps/shell', 'packages/ui'],
     },
   })
   ```

3. Create `apps/shell/vitest.config.ts`:

   ```typescript
   import { defineConfig, mergeConfig } from 'vitest/config'
   import viteConfig from './vite.config'

   export default mergeConfig(
     viteConfig,
     defineConfig({
       test: {
         environment: 'happy-dom',
         globals: true,
         include: ['src/**/*.test.ts'],
         setupFiles: ['./src/test-setup.ts'],
         coverage: {
           provider: 'v8',
           include: ['src/**/*.{ts,vue}'],
           exclude: [
             'src/main.ts',
             'src/**/*.d.ts',
             'src/data/**',
             'src/test-setup.ts',
           ],
           thresholds: {
             lines: 100,
             functions: 100,
             branches: 100,
             statements: 100,
           },
         },
       },
     }),
   )
   ```

4. Create `apps/shell/src/test-setup.ts`:

   ```typescript
   import { vi } from 'vitest'

   // Mock IntersectionObserver (not available in happy-dom)
   const MockIntersectionObserver = vi.fn().mockImplementation(() => ({
     observe: vi.fn(),
     unobserve: vi.fn(),
     disconnect: vi.fn(),
   }))
   vi.stubGlobal('IntersectionObserver', MockIntersectionObserver)

   // Clear localStorage between tests to prevent state leakage
   beforeEach(() => {
     localStorage.clear()
   })
   ```

5. Create `apps/shell/tsconfig.test.json`:

   ```json
   {
     "extends": "./tsconfig.app.json",
     "compilerOptions": {
       "types": ["vitest/globals"]
     },
     "include": ["src/**/*.test.ts", "src/test-setup.ts"]
   }
   ```

6. Create `packages/ui/vitest.config.ts`:

   ```typescript
   import { defineConfig } from 'vitest/config'
   import vue from '@vitejs/plugin-vue'
   import { resolve } from 'path'

   export default defineConfig({
     plugins: [vue()],
     resolve: {
       alias: {
         '@': resolve(__dirname, '../../apps/shell/src'),
         '@ui': resolve(__dirname, 'src'),
       },
     },
     test: {
       environment: 'happy-dom',
       globals: true,
       include: ['src/**/*.test.ts'],
       setupFiles: ['../../apps/shell/src/test-setup.ts'],
       coverage: {
         provider: 'v8',
         include: ['src/**/*.{ts,vue}'],
         thresholds: {
           lines: 100,
           functions: 100,
           branches: 100,
           statements: 100,
         },
       },
     },
   })
   ```

7. Add test scripts to root `package.json` scripts field:
   ```json
   "test": "vitest run",
   "test:watch": "vitest",
   "test:coverage": "vitest run --coverage"
   ```

Edit `.github/workflows/deploy.yml` to add a test step in the `build` job, between "Install dependencies" and "Build shell app":

```yaml
- name: Run tests with coverage
  run: bun run test:coverage
```

This step runs AFTER `bun install --frozen-lockfile` and BEFORE the `bun run build` step. If any test fails or coverage falls below 100%, the pipeline fails and deploy is blocked.

1. actions/checkout@v4
2. oven-sh/setup-bun@v2
3. Install dependencies (`bun install --frozen-lockfile`)
