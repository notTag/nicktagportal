**1. `apps/shell/src/composables/useTerminal.ts`** -- Vue composable for xterm.js lifecycle:

```typescript
import { onMounted, onUnmounted, type Ref, watch } from 'vue'
import { Terminal } from '@xterm/xterm'
import { FitAddon } from '@xterm/addon-fit'
import '@xterm/xterm/css/xterm.css' // CRITICAL: per Research Pitfall 6
import { Shell } from '@/terminal/Shell'
import { useTerminalThemeStore } from '@/stores/terminal'

export function useTerminal(containerRef: Ref<HTMLElement | null>) {
  const terminalThemeStore = useTerminalThemeStore()
  let terminal: Terminal | null = null
  let shell: Shell | null = null
  let fitAddon: FitAddon | null = null
  let resizeObserver: ResizeObserver | null = null

  onMounted(() => {
    if (!containerRef.value) return

    fitAddon = new FitAddon()

    terminal = new Terminal({
      fontSize: 14,
      lineHeight: 1.2,
      fontFamily: "'Menlo', 'DejaVu Sans Mono', 'Courier New', monospace",
      cursorBlink: true,
      cursorStyle: 'block',
      convertEol: true, // Per Research Pitfall 4
      scrollback: 1000,
      theme: { ...terminalThemeStore.xtermTheme }, // Spread per Pitfall 5
    })

    terminal.loadAddon(fitAddon)
    terminal.open(containerRef.value) // Per Pitfall 1: call in onMounted
    fitAddon.fit()

    // Click-anywhere focus per D-03
    containerRef.value.addEventListener('click', () => terminal?.focus())

    // Responsive resize per D-04 -- use ResizeObserver
    resizeObserver = new ResizeObserver(() => {
      if (terminal?.element) {
        // Guard per Pitfall 2
        fitAddon?.fit()
      }
    })
    resizeObserver.observe(containerRef.value)

    // Create and init Shell
    shell = new Shell(terminal, terminalThemeStore)
    shell.init()

    // Auto-focus terminal on mount
    terminal.focus()

    // Watch for external theme changes (if CLITHEME changed from outside)
    watch(
      () => terminalThemeStore.xtermTheme,
      (newTheme) => {
        if (terminal) {
          terminal.options.theme = { ...newTheme } // Spread per Pitfall 5
        }
      },
    )
  })

  onUnmounted(() => {
    // Disconnect ResizeObserver BEFORE dispose per Pitfall 2
    resizeObserver?.disconnect()
    resizeObserver = null
    shell?.dispose()
    shell = null
    terminal?.dispose()
    terminal = null
    fitAddon = null
  })
}
```

**2. `apps/shell/src/views/CliView.vue`** -- Complete rewrite per D-01, D-02, UI-SPEC layout contract:

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { useTerminal } from '@/composables/useTerminal'

const terminalContainer = ref<HTMLElement | null>(null)
useTerminal(terminalContainer)
</script>

<template>
  <div
    class="mx-auto flex max-w-5xl flex-col px-4 pt-6 pb-6"
    style="height: calc(100vh - 4rem)"
  >
    <div
      ref="terminalContainer"
      class="min-h-0 flex-1 overflow-hidden rounded-lg"
    />
  </div>
</template>
```

Layout rationale per D-01 and D-02:

- `max-w-5xl mx-auto px-4` matches site-width container per UI-SPEC layout contract
- `height: calc(100vh - 4rem)` accounts for header height (4rem / 64px) -- terminal fills remaining space per D-01
- `pt-6 pb-6` gives 24px vertical padding per UI-SPEC spacing scale (lg token)
- `flex-1 min-h-0` makes the terminal container fill available space within the flex column
- `overflow-hidden rounded-lg` contains xterm.js canvas with slight border radius per UI-SPEC spacing (sm = 8px)
- No `<h1>` heading -- the terminal IS the content, the banner serves as heading

The `ref="terminalContainer"` div is the xterm.js mount point. The composable calls `terminal.open(containerRef.value)` on this element.
